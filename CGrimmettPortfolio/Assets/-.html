Prime Number Check

The result of the prime check will appear here!

function primeNum() {
    var primeNum = Number(document.getElementById("prime").value);
    var isprime = [];
    for (i = 2; i < primeNum; i++) {
        if (primeNum % i == 0) {
            isprime.push(false);
        }
        else {
            isprime.push(true);
        }
    }
        if (isprime.includes(false)) {
            document.getElementById("primeresult").innerHTML = primeNum + " is NOT prime number";
        }
        else {
            document.getElementById("primeresult").innerHTML = primeNum + " IS a prime number";
        }
}
